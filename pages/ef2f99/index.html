Centos7配置静态网络

# centos7配置网络

# 1.查看服务器网络

ip addr     #查看服务器网络情况

# 2.修改相对应网口的配置文件

vi /etc/sysconfig/network-scripts/ifcfg-eno1     #打开eno1的配置文件,修改保存   

TYPE=Ethernet
PROXY_METHOD=none
BROWSER_ONLY=no
BOOTPROTO=static            #static静态、dhcp动态
DEFROUTE=yes
IPV4_FAILURE_FATAL=yes
IPADDR=192.168.3.123        #IP地址
NETMASK=255.255.255.0       #掩码
GATEWAY=192.168.3.1         #网关
DNS1=192.168.3.1            #DNS
IPV6INIT=yes
IPV6_AUTOCONF=yes
IPV6_DEFROUTE=yes
IPV6_FAILURE_FATAL=no
IPV6_ADDR_GEN_MODE=stable-privacy
NAME=eno1
UUID=212602a9-5702-4561-a402-37588f875fe8
DEVICE=eno1
ONBOOT=yes                  #自动启动

# 3.重启网卡服务

systemctl restart network           #重启网卡服务
ip addr                             #再次查看IP
