SpringCloud哨兵Sentinel

# 环境准备（sentinel）

sentinel下载 (opens new window)

# 启动

java -Dserver.port=8080 -Dcsp.sentinel.dashboard.server=localhost:8080 -Dproject.name=sentinel-dashboard -jar sentinel-dashboard.jar

# 新建sentinel哨兵模块

# 添加依赖

<dependencies>
    <!-- Sentinel哨兵 -->
    <dependency>
        <groupId>com.alibaba.cloud</groupId>
        <artifactId>spring-cloud-starter-alibaba-sentinel</artifactId>
    </dependency>

    <!-- 引入 SpringMVC 相关依赖，并实现对其的自动配置 -->
    <dependency>
        <groupId>org.springframework.boot</groupId>
        <artifactId>spring-boot-starter-web</artifactId>
    </dependency>
    
</dependencies>

# 配置

# yml配置

server:
  port: 8089
spring:
  application:
    name: sentinel
  cloud:
    nacos:
      discovery:
        server-addr: localhost:8848
    # sentinel 环境地址    
    sentinel:
      transport:
        dashboard: localhost:8080
      eager: true
      web-context-unify: true
      # 数据源 可自定义使用nacos、redis、file...
      datasource:
        ds1:
          file:
            file: classpath:flowrule.json
            data-type: json
            rule-type: flow

# 使用

# 启动类

@SpringBootApplication
@EnableDiscoveryClient
public class SentinelApplication {

    public static void main(String[] args) {
//        initFlowRules();
        SpringApplication.run(SentinelApplication.class);

    }

    /**
    * Java代码定义规则
    */
    private static void initFlowRules(){
        List<FlowRule> rules = new ArrayList<>(); //定义规则列表
        FlowRule rule = new FlowRule(); //定义一个规则
        rule.setResource("/hello"); //什么资源应用此规则
        rule.setGrade(RuleConstant.FLOW_GRADE_QPS);
        rule.setCount(2); //没秒只能2次请求
        rules.add(rule);//加入该规则
        FlowRuleManager.loadRules(rules); //载入规则列表
    }

}

# controller

@RestController
public class HelloController {

    @SentinelResource("resource")
    @GetMapping("/hello")
    public String hello() {
        return "Hello";
    }

    @GetMapping("/test")
    public String test() {
        return "test";
    }

}
